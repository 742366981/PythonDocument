"""
基于aiohttp的简单封装
"""
import asyncio

from aiohttp import web
from jinja2 import Environment, FileSystemLoader

BASE_DIR = ''


class WebApp:
    def __init__(self, host='127.0.0.1', port=8000, static_dir='static', template_dir='templates'):
        self.host =host
        self.port = port
        self.loop = asyncio.get_event_loop()
        self.app = web.Application(loop=self.loop)
        self.static = BASE_DIR + '/' + static_dir
        self.template = Template(BASE_DIR + '/' + template_dir)

    async def init(self):
        runner = web.AppRunner(self.app)
        await runner.setup()
        site = web.TCPSite(runner, self.host, self.port)
        print(f'Server started at http://{self.host}:{self.port}')
        await site.start()

    def start(self):
        self.loop.run_until_complete(self.init())
        self.loop.run_forever()

    def add_route(self, method, url, handler):
        self.app.router.add_route(method, url, handler)

    def add_routes(self, routes):
        for route in routes:
            self.add_route(route[0], route[1], route[2])

    def get_response(self, template_name, data,content_type='text/html'):
        return web.Response(text=self.template.render(template_name, data), content_type=content_type, charset='utf-8')


class Template:
    def __init__(self, template_dir):
        self.template_dir = template_dir
        self.environment = Environment(loader=FileSystemLoader(self.template_dir))

    def render(self, template_name, data):
        return self.environment.get_template(template_name).render(**data)

"""
基于logging的封装
"""
import logging
import os


class Logger:
    CRITICAL = 50
    FATAL = CRITICAL
    ERROR = 40
    WARNING = 30
    WARN = WARNING
    INFO = 20
    DEBUG = 10
    NOTSET = 0

    def __init__(self, name, folder):
        """
        :param name: 日志文件前缀名
        :param folder: 日志所在文件夹
        """
        self.name = name
        self.log_folder = os.path.abspath(folder)
        if not os.path.exists(self.log_folder):
            os.mkdir(self.log_folder)
        self.formatter = logging.Formatter('%(asctime)s %(name)s- %(levelname)s - %(message)s')

    def save_log(self, msg, level):
        if not hasattr(self, f'logger_{level}'):
            file_handler = logging.FileHandler(f'{self.log_folder}/{self.name}_{level}.log',encoding='utf-8')
            file_handler.setLevel(eval(f'self.{level.upper()}'))
            file_handler.setFormatter(self.formatter)
            logger = logging.getLogger(self.name + level)
            logger.setLevel(eval(f'self.{level.upper()}'))
            logger.addHandler(file_handler)
            setattr(self, f'logger_{level}', logger)
        handlers_length = len(eval(f'self.logger_{level}.handlers'))
        if  handlers_length > 1:
            for i in range(handlers_length - 1):
                eval(f'self.logger_{level}.handlers.pop()')
        eval(f'self.logger_{level}.{level}({repr(msg)})')

    def save_critical(self, msg, level='critical'):
        self.save_log(msg, level)

    def save_error(self, msg, level='error'):
        self.save_log(msg, level)

    def save_warning(self, msg, level='warning'):
        self.save_log(msg, level)

    def save_info(self, msg, level='info'):
        self.save_log(msg, level)

    def save_debug(self, msg, level='debug'):
        self.save_log(msg, level)
        
        
web_log = Logger('web', 'logs')
script_log = Logger('script', 'logs')
