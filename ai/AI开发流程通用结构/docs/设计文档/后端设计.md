## 2. 应用工厂

核心文件：`apps/__init__.py`

### 2.1 初始化流程

```python
def create_app(protect_swagger=True):
    app = Flask(__name__)

    # 1. 配置加载
    app.config['SECRET_KEY'] = app_conf.get('secret_key')
    app.config['DEBUG'] = app_conf.get('debug').lower() == 'true'

    # 2. 数据库配置
    app.config['SQLALCHEMY_DATABASE_URI'] = f'mysql+pymysql://...'

    # 3. 初始化扩展
    db.init_app(app)
    migrate.init_app(app, db)
    compress.init_app(app)

    # 4. CORS配置
    CORS(app, resources={r'/*': {'origins': '*', 'supports_credentials': True}})

    # 5. 中间件
    init_request_id(app)
    init_request_log(app)

    # 6. 注册蓝图
    register_blueprints(app)

    # 7. 注册异常处理
    register_error_handlers(app)

    # 8. 注册Swagger（仅开发环境）
    register_swagger(app, protect=protect_swagger)

    return app
```
