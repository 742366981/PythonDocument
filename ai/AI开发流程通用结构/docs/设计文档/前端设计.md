### 2.2 API 方法标准模板

```javascript
export const baseApi = {
  moduleName: {
    list: (params) => api.get('/module/list', params),
    detail: (id) => api.get('/module/detail', { id }),
    create: (data) => api.post('/module/create', data),
    update: (data) => api.post('/module/update', data),
    delete: (id) => api.post('/module/delete', { id }),
    batchDelete: (ids) => api.post('/module/batch-delete', { ids }),
    import: (formData) => axiosInstance.post('/module/import', formData, {
      headers: { 'Content-Type': 'multipart/form-data' }
    }),
    export: () => downloadBlob('/module/export'),
    template: () => downloadBlob('/module/template/download', {}, 'GET')
  }
}
```

## 5. 状态管理

### 5.1 Auth Store

```javascript
export const useAuthStore = defineStore('auth', () => {
  const token = ref(localStorage.getItem('token') || '')
  const userInfo = ref(
    localStorage.getItem('userInfo')
      ? JSON.parse(localStorage.getItem('userInfo'))
      : null
  )

  const isLoggedIn = computed(() => !!token.value)

  function logout() {
    token.value = ''
    userInfo.value = null
    localStorage.removeItem('token')
    localStorage.removeItem('userInfo')
  }

  return { token, userInfo, isLoggedIn, logout }
})
```
