### 2.3 配置加载

```python
# common/confs.py

import configparser
import os
from common.constants import BASE_DIR, IS_PRODUCT, IS_OUTER

if IS_PRODUCT:
    config_dir = f'{BASE_DIR}/config'
else:
    config_dir = f'{BASE_DIR}/config/test'

if IS_OUTER:
    config_path = f'{config_dir}/config_outer.ini'
else:
    config_path = f'{config_dir}/config.ini'

def read_config_file(file_path, section):
    if not os.path.exists(file_path):
        raise GeneralError(f'配置文件<{file_path}>不存在')
    config = configparser.ConfigParser()
    config.read(file_path, encoding='utf-8')
    if not config.has_section(section):
        raise GeneralError(f'配置文件<{file_path}>，节点<{section}>不存在')
    return dict(config.items(section))

redis_conf = read_config_file(config_path, 'redis')
service_api_conf = read_config_file(config_path, 'callback_service')
```

### 2.4 启动参数

启动参数通过 `sys.argv` 获取，**不写在配置文件中**：

```python
# common/constants.py

import os
import sys

PLATFORM = os.name

IS_PRODUCT = True
IS_OUTER = False
IS_ONCE = False

sys_args = sys.argv[1:]
if sys_args:
    if '--test' in sys_args:
        IS_PRODUCT = False
    if '--outer' in sys_args:
        IS_OUTER = True
    if '--once' in sys_args:
        IS_ONCE = True

# 运行类型（第1个参数）
RUN_TYPE = sys_args[0] if len(sys_args) > 0 else 'default'

# 线程数（第2个参数）
RUN_THREAD_SIZE = int(sys_args[1]) if len(sys_args) > 1 else 1

# 数据源ID（第3个参数）
DATASOURCE_ID = sys_args[2] if len(sys_args) > 2 else 'DEFAULT'
```

**启动参数说明：**

| 参数位置 | 变量名 | 说明 | 示例 |
|:---------|:-------|:-----|:-----|
| sys_args[0] | RUN_TYPE | 运行类型 | review、seize、data |
| sys_args[1] | RUN_THREAD_SIZE | 线程数 | 1、4、8 |
| sys_args[2] | DATASOURCE_ID | 数据源标识 | HU310、BK001 |

## 4. 通用函数分层

### 4.1 分层设计

爬虫函数分为 **通用层** 和 **模块层**：

| 层级 | 位置 | 职责 |
|:-----|:-----|:-----|
| 通用层 | apps/general/ | 跨模块复用：Session/代理/浏览器/请求封装等 |
| 模块层 | apps/{module}/ | 业务逻辑：具体业务的执行/解析/回调等 |

### 4.2 通用层模块（apps/general/）

| 文件 | 职责 |
|:-----|:-----|
| crawl_constants.py | 通用常量（Browser指纹、Redis Key前缀） |
| crawl_functions.py | 通用函数（Session获取、代理管理、请求封装等） |
| crawl_locks.py | 文件锁定义 |
| crawl_retry_conditions.py | 通用重试条件 |
| schedulers.py | 通用调度器 |

### 4.3 模块层模块（apps/{module}/）

每个业务模块实现具体的：
- `run_xxx()` - 业务执行
- `parse_xxx()` - 数据解析
- `callback_xxx()` - 结果回调

### 4.4 通用Session获取

```python
# apps/general/crawl_functions.py

def get_valid_session(ip=None, need_activate=False):
    """获取有效的Session（带Cookie）"""
    while True:
        request_args = redis_crawl.rpop(REDIS_KEY_LIST_XXX_REQUEST_ARGS)
        if not request_args:
            break
        request_args = json.loads(request_args)
        now_ts = get_now_ts()
        create_time_ts = yyyy_mm_dd_s_to_ts(request_args['create_time'])
        if now_ts - create_time_ts >= 180:  # 超过3分钟则丢弃
            continue
        break

    if request_args:
        session = get_proxy_session(ip=ip)
        session.cookies = cookiejar_from_dict(request_args['cookies'])
        session.request_args = request_args
        return session

    # 创建新Session
    session = get_init_session(ip, need_activate)
    request_args = {
        'create_time': ts_to_yyyy_mm_dd_s(),
        'cookies': {
            'JSESSIONID': session.cookies.get('JSESSIONID'),
            'route': session.cookies.get('route')
        }
    }
    session.request_args = request_args
    return session
```

| 类型 | 实现方式 | 示例 |
|:-----|:---------|:-----|

### 6.2 任务参数结构

每个任务包含 **执行参数** 和 **extra 上下文**：

```python
# 任务参数示例
args = (flight_date, dep_code, arr_code, flight_number, cabin_code_list, extra)

# extra 上下文包含
extra = {
    'id': 12345,              # 任务唯一标识（用于分布式锁）
    'request_args': {},       # 请求参数（会话复用）
    'cache_data': {},         # 缓存数据（用于断点续执）
    'execute_time': '',       # 执行开始时间
    'complete_time': '',      # 执行结束时间
    # ... 其他自定义字段
}
```

### 6.3 Redis队列任务源实现

```python
def xxx_query_task_args_list(count):
    # 1. 尝试从Redis队列获取任务
    db_data = redis_crawl.lpop(REDIS_KEY_LIST_DISTRIBUTE_TASK_XXX, count)
    if db_data:
        task_args_list = [tuple(json.loads(obj)) for obj in db_data]
        if len(task_args_list) > 0:
            return task_args_list

    # 2. 分布式锁保护，从后端API获取
    lock = redis_crawl.atomic_lock(REDIS_KEY_STRING_LOCK_DISTRIBUTE_TASK_XXX, ex=60)
    if not lock:
        return []

    # 3. 从后端API分页获取任务
    task_args_list = xxx_query_task_args_chunk_size(1, 1000)

    # 4. 写入Redis队列供后续消费
    redis_task_args_list = [json.dumps(args) for args in task_args_list]
    if len(task_args_list) > 0:
        redis_crawl.rpush(REDIS_KEY_LIST_DISTRIBUTE_TASK_XXX, *redis_task_args_list)

    redis_crawl.delete(REDIS_KEY_STRING_LOCK_DISTRIBUTE_TASK_XXX)
    return []
```

### 6.4 API任务源实现

```python
def xxx_query_task_args_chunk_size(page_no, page_size):
    host = service_api_conf['callback_service']
    url = f'{host}/api/{{project}}/getTasks'
    data = {'datasource_id': DATASOURCE_ID, 'page_no': page_no, 'page_size': page_size}
    with requests_try('post', url, json=data) as resp:
        if type(resp) is str:
            raise GeneralError(resp)
        resp_data = resp.json()
        if resp_data['code'] != 200:
            raise GeneralError(f'任务获取失败：{json.dumps(resp_data)}')

    task_args_list = []
    for record in resp_data['data']['records']:
        # 过滤和转换任务数据
        uid = f'{record["id"]}'
        # 任务锁检查
        if redis_crawl.get(f'{REDIS_KEY_STRING_LOCK_TASK_XXX}{uid}'):
            continue
        extra = {'id': record['id'], 'cache_data': json.loads(record['cache_data']) if record.get('cache_data') else None}
        args = (flight_date, dep_code, arr_code, flight_number, cabin_code_list, extra)
        task_args_list.append(args)
    return task_args_list
```

## 7. 结果存储（任务回调）

### 7.1 存储类型

| 类型 | 实现方式 | 示例 |
|:-----|:---------|:-----|
| 本地文件 | JSON/Excel文件 | xxx_callback_save_json() |
| Redis缓存 | 结果缓存 | xxx_callback_redis() |
| 后端API | HTTP提交 | xxx_callback_api() |
| 数据库 | 写入数据库 | xxx_callback_save_db() |
| 消息队列 | 发送MQ消息 | xxx_callback_send_mq() |

### 7.2 多级回调机制

通常需要组合多种回调方式，例如 **Redis缓存 + API回调**：

```python
def xxx_task_wrapper(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        extra = args[-1]
        try:
            # 1. 任务分布式锁
            uid = f'{extra["id"]}'
            lock = redis_crawl.atomic_lock(f'{REDIS_KEY_STRING_LOCK_TASK_XXX}{uid}', ex=120)
            if not lock:
                return

            # 2. 执行任务
            result = func(*args, **kwargs)

            # 3. Redis缓存回调
            xxx_callback_success_redis(result, extra)

            # 4. 后端API回调
            xxx_callback_success(*result, extra)

            # 5. 删除任务锁
            redis_crawl.delete(f'{REDIS_KEY_STRING_LOCK_TASK_XXX}{uid}')
            return result

        except Exception as e:
            # 6. 失败处理
            xxx_callback_fail_redis(hint, extra)
            xxx_callback_fail(hint, extra)
            redis_crawl.delete(f'{REDIS_KEY_STRING_LOCK_TASK_XXX}{uid}')
            return hint
    return wrapper
```

### 7.3 Redis缓存回调

```python
def xxx_callback_success_redis(result, extra):
    """结果存入Redis供后续查询"""
    # result 为元组，包含执行结果和缓存数据
    # result = (result_data, ..., cache_data)

from utils.contexts import requests_try, requests_session_try

# 独立请求

### 9.2 重试条件函数

def review_retry_if_exception(e):
    hint = str(e)
    if check_hint_keywords(hint, ['代理ip失效', '接口超时', '接口连接错误', '状态码异常']):
        return True  # 可重试
    return False  # 不重试
```

### 9.3 可重试错误关键词

| 关键词 | 说明 |
|:--------|:-----|
| 代理ip失效 | 代理不可用 |
| 接口超时 | 请求超时 |
| 接口连接错误 | 连接失败 |
| HTTP错误 | HTTP请求错误 |
| 状态码异常 | 非200状态码 |

class GeneralError(Exception):
    def __str__(self):

        if not PROGRAM_SECURE_SWITCH:

## 12. 任务执行流程

### 12.1 任务生成器

```python
def review_task_generator():
    """
    从任务源获取任务，分发给执行器

    任务源可以是：Excel、API、数据库、消息队列等
    """
    task_args_list = review_query_task_args_list()  # 获取任务源
    for args in task_args_list:
        yield review_crawl_run, args
```

### 12.2 任务执行循环

```python
def review_run_task_loop(thread_pool):
    task_generator = review_task_generator()
    while True:
        try:
            fn, args = next(task_generator)
            thread_pool.submit(fn, *args)
        except StopIteration:
            break
    thread_pool.run_until_complete()
```

## 13. 任务包装器

### 13.1 包装器职责

```python
def xxx_task_wrapper(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        extra = args[-1]
        uid = f'{extra["id"]}'

        try:
            # 1. 任务运行标志检查
            if not check_xxx_run_flag(extra['id']):
                return

            # 2. 任务分布式锁
            lock = redis_crawl.atomic_lock(f'{REDIS_KEY_STRING_LOCK_TASK_XXX}{uid}', ex=120)
            if not lock:
                return

            # 3. 记录执行时间
            extra['execute_time'] = ts_to_yyyy_mm_dd_s()

            # 4. 执行任务
            result = func(*args, **kwargs)

            # 5. Redis缓存回调
            xxx_callback_success_redis(result, extra)

            # 6. 记录完成时间
            extra['complete_time'] = ts_to_yyyy_mm_dd_s()
            extra['duration'] = yyyy_mm_dd_s_to_ts(extra['complete_time']) - yyyy_mm_dd_s_to_ts(extra['execute_time'])

            # 7. 后端API回调
            extra['callback_execute_time'] = ts_to_yyyy_mm_dd_s()
            xxx_callback_success(*result, extra)
            extra['callback_complete_time'] = ts_to_yyyy_mm_dd_s()
            extra['callback_duration'] = yyyy_mm_dd_s_to_ts(extra['callback_complete_time']) - yyyy_mm_dd_s_to_ts(extra['callback_execute_time'])

            # 8. 删除任务锁
            redis_crawl.delete(f'{REDIS_KEY_STRING_LOCK_TASK_XXX}{uid}')

            # 9. 记录结果日志
            msg = f'任务类型：{XXX_TASK_TITLE}，参数：{args}，执行完毕，爬虫耗时Xs，回调耗时Ys'
            xxx_result_log.save_info(msg)
            return result

        except Exception as e:
            # 10. 分类处理异常
            hint_repr = repr(e)
            hint = str(e)
            if isinstance(e, GeneralError):
                xxx_error_log.save_error(f'***任务执行异常捕获*** {hint_repr}')
            elif isinstance(e, NeedDingtalkReminderError):
                reminder_dingtalk.send_text_msg(...)
            else:
                xxx_error_log.save_critical(f'***任务执行异常捕获*** {traceback_str}')

            # 11. 失败回调
            xxx_callback_fail_redis(hint, extra)
            if extra.get('cache_data'):
                xxx_callback_fail(hint, extra)

            redis_crawl.delete(f'{REDIS_KEY_STRING_LOCK_TASK_XXX}{uid}')
            return hint

    return wrapper
```

- 工作目录：`/{{project_name}}`

- 可按需安装 Node.js（某些爬虫项目需要）

### 23.2 docker-compose.yml

```yaml
version: '2.1'
services:
    {{project_name}}-{{module}}-{{datasource_id}}-1:
            build:
                context: ./
                dockerfile: Dockerfile
            image: python-{{project_name}}
            container_name: {{project_name}}-{{module}}-{{datasource_id}}-1
            working_dir: /{{project_name}}
            restart: always
            command: python3 -u main.py {{module}} {{thread_size}} {{datasource_id}}
            volumes:
                - ../{{project_name}}:/{{project_name}}
            environment:
                - TZ=Asia/Chongqing
            logging:
                driver: 'json-file'
                options:
                    max-size: '100m'
                    max-file: '2'
```

**要点：**
- 一个 docker-compose 运行**一个线程**
- 多个线程需要配置**多个 service**（如 `-1`、`-2`、`-3`...）
- command 格式：`python3 -u main.py <module> <thread_size> <datasource_id>`
- volumes 挂载代码目录，实现代码更新后容器内同步
- logging 控制日志大小和文件数量
